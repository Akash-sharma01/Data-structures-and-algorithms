#include<conio.h>
#include<iostream>
using namespace std;
int main(){
	int n; cin>>n;
	int arr[n];
	for(int i=0; i<n; i++){
		cin>>arr[i];
	}
	int temp[n];
	int j=0;
    for(int i=0; i+1<n; i++){
	    if(arr[i] != arr[i+1] ){
		   temp[j] = arr[i];             
		   j++;
        }  
    }
    temp[j] = arr[n-1];
    j++;
    for(int i=0; i<j; i++){
	cout<<temp[i]<<" ";
    }   
    getch();
}
